    ```shell script
    sudo apt install python3-nautilus python3-psutil python3-pip libglib2.0-bin dconf-editor
    # user install
    pip3 install --user nautilus_terminal
    # system wide install
    sudo pip3 install nautilus_terminal
    # kill nautilus to load new version
    nautilus -q
    <F4>: toggle
    ```
